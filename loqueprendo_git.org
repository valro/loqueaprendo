* Configurar los parámetros iniciales

#+BEGIN_SRC sh

git config --global user.name "Nombre que quieras mostrar"
git config --gloabal user.email "correo@electronico.com"

#+END_SRC

  2. Ejecutar:

     #+BEGIN_SRC sh

     git init
